    document.write(obj.b[0]);
    document.write(obj.b[1]);
    document.write(obj.c);
    obj.d();
    obj.e();
    obj.f();
}

17. 변수의 종류 : 지역변수, 전역변수

{
        document.write("*** 17. 변수의 종류 *** ");

        let x = 100;    //변수 설정 : 전역변수
        let y = 200;
        let z = "javascript";
        let a = 300;
        
        //선언적 함수
        function func(){
            let x = 100;    //변수 설정 : 지역변수
            let y = 200;
            let z = "javascript";
            a = 400;        // 전역 변수 : 변수 값이 변경
            y = 500;        // 지역 변수 : 변수 값이 200 -> 500으로 변경

            document.write("함수 안");
            document.write(x);
            document.write(y);
            document.write(z);
            document.write(a);
        }
        func();

        document.write("함수 밖");
        document.write(x);
        document.write(y);
        document.write(z);
        document.write(a);
}

18. 변수의 변환 : 형 변환

{
        ocument.write("*** 18. 변수의 변환 *** ");

        let x = 100;
        let y = "100";

        document.write(x + ":" + typeof(x));
        document.write(y + ":" + typeof(y));

        x = "" + x; // 숫자를 문자열로 형 변환
        document.write(x + ":" + typeof(x));

        y = Number(y);  // 내장 객체를 이용한 형 변환
        document.write(y + ":" + typeof(y));

        x = Number(x);
        document.write(x + ":" + typeof(x));

        y = String(y);
        document.write(y + ":" + typeof(y));
}
