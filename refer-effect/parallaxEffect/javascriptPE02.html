 document.querySelectorAll(".paraDot ul li a").forEach((element) =>  {
    element.addEventListener("click" , (e) => {
        e.preventDefault();
        document.querySelector(element.getAttribute("href")).scrollIntoView({
            behavior : "smooth"
        });
    });
});

window.addEventListener("scroll" , () => {
    let scrollTop = document.documentElement.scrollTop || window.scrollY || window.pageYOffset
    document.querySelector(".paraScroll span").innerText = parseInt(scrollTop) + "px";

    // if( scrollTop > document.getElementById("section1").offsetTop -2){
    //     document.querySelectorAll(".paraDot ul li").forEach((el) => {
    //         el.classList.remove("active");
    //     })
    //     document.querySelector(".paraDot ul li:nth-child(1)").classList.add("active")
    // }

    document.querySelectorAll('.paraDot ul li').forEach((li, idx) => {
        if (scrollTop > document.getElementById('section' + (idx + 1)).offsetTop - 2) {
            document.querySelectorAll('.paraDot ul li').forEach(el => {
                el.classList.remove('active');
        });
    document.querySelector('.paraDot ul li:nth-child(' + (idx + 1) + ')').classList.add('active');
    }
    });
})
