$(window).mousemove(function(e){
    let pageX = e.pageX,
        pageY = e.pageY;

    //전체 가로 세로  : 기준점을 가운데로 변경
    let standardX = $(window).width()/2 - pageX;
    let standardY = $(window).height()/2 - pageY;

    //최대값, 최소값 설정
    // if(standardX >= 50)  standardX = 50
    // if(standardX <= -50)  standardY = -50
    // if(standardY >= 50)  standardY = 50
    // if(standardY <= -50)  standardY = -50


    //최소값은 -50 최대값은 50 설정
    let maxX = Math.max(-100, Math.min(100, standardX)); // standardX 의 값이 넘어갈때 50으로 제한 -> 최대값
    let maxY = Math.max(-100, Math.min(100, standardY));

    //각도를 줄이는 설정
    let angleX = maxX * 0.1;
    let angleY = maxY * 0.1;

    //부드럽게 설정
    let softX = 0;
    let softY = 0;

    softX += (angleX - softX) * 0.8;
    softY += (angleY - softY) * 0.8;

    //이미지
    $(".moveImage").css({"transform": "translate(-50% , -50%) perspective(600px) rotateX("+ -softX+"deg) rotateY("+ -softY+"deg)"});

    //커서
    gsap.to(".cursor" , {duration: .3, left: pageX , top : pageY});

    //출력
    $(".pageX").text(pageX);
    $(".pageY").text(pageY);
    $(".standardX").text(standardX);
    $(".standardY").text(standardY);
    $(".maxX").text(maxX);
    $(".maxY").text(maxY);
    $(".angleX").text(parseInt(angleX));
    $(".angleY").text(parseInt(angleY));
});
