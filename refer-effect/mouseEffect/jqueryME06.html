$(window).mousemove(function(e){
    let standardX = $(".cursor").width()/2;
    let standardY = $(".cursor").height()/2;

    gsap.to(".cursor" , {duration: 0.5, left : e.pageX - standardX , top: e.pageY - standardY });
})
