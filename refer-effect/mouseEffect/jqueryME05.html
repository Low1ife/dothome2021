.moveWrap {
    display: flex;
    align-items: center;
    justify-content: center;
    width: 100%; height: 100vh;
}
.moveImage {
    position: relative;
}
.moveImage .mImg {
    width: 50vw;
    height: 60vh;
    overflow: hidden;
    position: relative;
    transition : transform 500ms cubic-bezier(0.25, 0.46, 0.45, 0.84);
}
.moveImage:hover .mImg {
    transform: scale(1.025);
}
.moveImage .mImg img {
    width: 100%;
    position: absolute;
    left: -5%; top: -5%;
    width: 110%;
    height: 110%;
    object-fit: cover;
    transition : transform 500ms cubic-bezier(0.25, 0.46, 0.45, 0.84);
}
.moveImage .mText {
    position: absolute;
    left: 50%; top: 50%;
    transform : translate(-50% , -50%);
    text-align: center;
    font-size: 1.3vw;
    white-space: nowrap;
    color : #fff;

.cursor {
    position: absolute;
    width: 20px; height: 20px;
    background: #fff;
    border-radius: 50%;
    z-index: 1000;
    user-select: none;
    pointer-events: none;
}

$(".moveImage").mousemove(function(e){
    let pageX = e.pageX,
        pageY = e.pageY;
    gsap.to(".cursor" , {duration : .2, left: e.pageX -10 , top : e.pageY -10})

    //1. 좌표값 가운데 정렬
    let standardX = $(window).width()/2 -pageX;
    let standardY = $(window).height()/2  -pageY;

    //2. img x -> 마우스 값 / 20 y -> 마우스 값
    $(".mImg img").attr({"style": "transform : translate("+standardX/20+"px, "+standardY/20 + "px)"});


    $(".pageX").text(pageX);
    $(".pageY").text(pageY);
    $(".standardX").text(standardX);
    $(".standardY").text(standardY);
})
