 const tabView = $(".view-title ul li")
const tabCovi = $(".view-cont > div")

tabView.click(function () {
    let target = $(this);
    tabView.removeClass("active");
    target.addClass("active");

    let index = target.index();
    tabCovi.css("display", "none");
    tabCovi.eq(index).css("display", "block");
});
