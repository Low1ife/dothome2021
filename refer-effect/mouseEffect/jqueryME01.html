// 출력문
$(window).mousemove(function(e){
    $(".clientX").text(e.clientX);  // 브라우저 X 좌 표 기준
    $(".clientY").text(e.clientY);  // 브라우저 Y 좌 표 기준
    $(".offsetX").text(e.offsetX);  // 요소 좌표
    $(".offsetY").text(e.offsetY);  // 요소 Y 좌표
    $(".pageX").text(e.pageX);      // 페이지 X 좌표
    $(".pageY").text(e.pageY);
    $(".screenX").text(e.screenX);
    $(".screenY").text(e.screenY);
});
